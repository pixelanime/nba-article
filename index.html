Add Apps Script

Go to Extensions → Apps Script in your Sheet. Delete the default code and paste this:

Code.gs — Google Apps Script 
function doGet(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Articles');
    if (!sheet) return jsonResponse({ error: 'Sheet "Articles" not found' }, 404);
    var data = sheet.getDataRange().getValues();
    if (data.length <= 1) return jsonResponse([]);
    var headers = data[0];
    var result = [];
    for (var i = 1; i < data.length; i++) {
      var row = {};
      for (var j = 0; j < headers.length; j++) {
        row[headers[j].toLowerCase()] = data[i][j];
      }
      result.push(row);
    }
    return jsonResponse(result);
  } catch (err) {
    return jsonResponse({ error: err.message }, 500);
  }
}

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Articles');
    if (!sheet) return jsonResponse({ error: 'Sheet "Articles" not found' }, 404);

    var payload = JSON.parse(e.postData.contents);

    // Simple auth check
    var adminKey = PropertiesService.getScriptProperties().getProperty('ADMIN_KEY') || 'changeme123';
    if (payload.admin_key !== adminKey) {
      return jsonResponse({ error: 'Unauthorized' }, 401);
    }

    var date = payload.date || new Date().toISOString().split('T')[0];
    sheet.appendRow([payload.title, payload.slug, payload.category, payload.content, payload.image_url, date]);
    return jsonResponse({ success: true, message: 'Article added successfully' });
  } catch (err) {
    return jsonResponse({ error: err.message }, 500);
  }
}

function jsonResponse(data, code) {
  code = code || 200;
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON)
    .setContentLength(JSON.stringify(data).length);
}

// Run this once to set your admin password
function setAdminKey() {
  PropertiesService.getScriptProperties().setProperty('ADMIN_KEY', 'your-secure-password-here');
}
3

Deploy as Web App

1. Click Deploy → New Deployment

2. Set type to Web App

3. Execute as: Me

4. Who has access: Anyone

5. Click Deploy, authorize, and copy the Web App URL

4

Connect This Dashboard

Paste your Web App URL and admin password in the Settings panel (gear icon in the sidebar), then you're ready to publish.

Security Tip: Change the default admin password by running setAdminKey() in Apps Script with your own password before deploying.
